 #include <iostream>
 using namespace std;
 int main() {
     int area = 0, cnt = 0; //面积area、计数变量cnt
     cin >> area; //输入面积
     for (int w = 1; w * w <= area; w++) { //宽从1 开始到宽*宽<=面积
         if (area % w == 0) //如果面积是宽的倍数，说明满足题意，cnt++
            cnt++;
     }
     cout << cnt << endl; //输出cnt
     return 0;
 }
