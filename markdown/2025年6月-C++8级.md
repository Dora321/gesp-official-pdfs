    int cnt = 0;
    for (int v = 1; v * v * 4 <= n; v++)
        for (int u = v + 1; u * (u + v) * 2 <= n; u += 2)
            if (gcd(u, v) == 1) {

int n, q;
int h[N], nx[N];
int par[L][N], son[L][N];

void dfs(int u) {
    for (int i = 1; i < L; i++)
        par[i][u] = par[i - 1][par[i - 1][u]];
    son[0][u] = u;
    for (int i = h[u]; i; i = nx[i]) {
        dfs(i);
        if (son[0][u] == u || i < son[0][u])
            son[0][u] = i;
    }
    for (int i = 1; i < L; i++)
        son[i][u] = son[i - 1][son[i - 1][u]];
}

int move(int go[L][N], int u, int step) {
    for (int i = 0; i < L; i++)
        if ((step >> i) & 1)
            u = go[i][u];
    return u;
}
