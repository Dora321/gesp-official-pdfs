int n;
int l[N], r[N];
int mn[N], mx[N], chk[N];
int ans;
void dfs(int u) {
    chk[u] = 1;
    if (!u)
        return;
    dfs(l[u]);
    dfs(r[u]);
    chk[u] &= chk[l[u]] & chk[r[u]];
    mn[u] = 1 + min(mn[l[u]], mn[r[u]]);
    mx[u] = 1 + max(mx[l[u]], mx[r[u]]);
    chk[u] &= mn[l[u]] >= mx[r[u]];
    chk[u] &= mn[u] >= mx[u] - 1;
    ans += chk[u];
}
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i++)
        scanf("%d%d", &l[i], &r[i]);
    dfs(1);
    printf("%d\n", ans);
    return 0;
}
