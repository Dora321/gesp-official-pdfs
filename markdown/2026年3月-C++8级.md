        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= n; j++)
                g[i][j] = f[i][j];
        for (int r = l; r <= n; r++) {
            for (int i = 1; i <= n; i++)
                for (int j = 1; j <= n; j++)
                    g[i][j] = min(g[i][j], g[i][r] + g[r][j]);
            for (int i = l; i <= r; i++)
                for (int j = i; j <= r; j++)
                    ans = (ans + g[i][j]) % mod;
        }
    }
    printf("%d\n", ans);
    return 0;
}
