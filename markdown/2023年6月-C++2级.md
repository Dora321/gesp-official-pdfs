#include <iostream>
using namespace std;
int main() {
    int m = 0;
    cin >> m;
    for (int i = 0; i < m; i++) {
        int n = 0;
        cin >> n;
        // 数一下n 有多少位数，记为l
        int t = n, l = 0;
        while (t > 0) {
            t /= 10;
           l++;
        }
        // 每位数l 次方求和，记为sum
        int sum = 0;
        t = n;
        while (t > 0) {
           int d = t % 10;
            t /= 10;
           int mul = 1;
           for (int j = 0; j < l; j++)
               mul *= d;
           sum += mul;
        }
        // 根据sum 和n 是否相等，判断是否为自幂数
        if (sum == n)
           cout << "T" << endl;
        else
           cout << "F" << endl;
    }
    return 0;
}
