#include <iostream>
using namespace std;
int main() {
    int n;
    cin >> n;
    int sum = 0; // 定义计数变量sum 初始化为0，用于累加求和
    for (int i = 1; i <= n; i++) // 遍历每一项，i 表示当前项数
        sum += (i + 1) * i / 2; // 把每个多项式的值累加到sum 中求和

}
